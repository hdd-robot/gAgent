30/** \brief Initialize a UDP client object.
31 *
32 * This function initializes the UDP client object using the address and the
33 * port as specified.
34 *
35 * The port is expected to be a host side port number (i.e. 59200).
36 *
37 * The \p addr parameter is a textual address. It may be an IPv4 or IPv6
38 * address and it can represent a host name or an address defined with
39 * just numbers. If the address cannot be resolved then an error occurs
40 * and constructor throws.
41 *
42 * \note
43 * The socket is open in this process. If you fork() or exec() then the
44 * socket will be closed by the operating system.
45 *
46 * \warning
47 * We only make use of the first address found by getaddrinfo(). All
48 * the other addresses are ignored.
49 *
50 * \exception udp_client_server_runtime_error
51 * The server could not be initialized properly. Either the address cannot be
52 * resolved, the port is incompatible or not available, or the socket could
53 * not be created.
54 *
55 * \param[in] addr The address to convert to a numeric IP.
56 * \param[in] port The port number.
57 */
58udp_client::udp_client(const std::string& addr, int port)
59 : f_port(port)
60 , f_addr(addr)
61{
62 char decimal_port[16];
63 snprintf(decimal_port, sizeof(decimal_port), "%d", f_port);
64 decimal_port[sizeof(decimal_port) / sizeof(decimal_port[0]) - 1] = '\0';
65 struct addrinfo hints;
66 memset(&hints, 0, sizeof(hints));
67 hints.ai_family = AF_UNSPEC;
68 hints.ai_socktype = SOCK_DGRAM;
69 hints.ai_protocol = IPPROTO_UDP;
70 int r(getaddrinfo(addr.c_str(), decimal_port, &hints, &f_addrinfo));
71 if(r != 0 || f_addrinfo == NULL)
72 {
73 throw udp_client_server_runtime_error(("invalid address or port: \"" + addr + ":" + decimal_port + "\"").c_str());
74 }
75 f_socket = socket(f_addrinfo->ai_family, SOCK_DGRAM | SOCK_CLOEXEC, IPPROTO_UDP);
76 if(f_socket == -1)
77 {
78 freeaddrinfo(f_addrinfo);
79 throw udp_client_server_runtime_error(("could not create socket for: \"" + addr + ":" + decimal_port + "\"").c_str());
80 }
81}

133/** \brief Send a message through this UDP client.
134 *
135 * This function sends \p msg through the UDP client socket. The function
136 * cannot be used to change the destination as it was defined when creating
137 * the udp_client object.
138 *
139 * The size must be small enough for the message to fit. In most cases we
140 * use these in Snap! to send very small signals (i.e. 4 bytes commands.)
141 * Any data we would want to share remains in the Cassandra database so
142 * that way we can avoid losing it because of a UDP message.
143 *
144 * \param[in] msg The message to send.
145 * \param[in] size The number of bytes representing this message.
146 *
147 * \return -1 if an error occurs, otherwise the number of bytes sent. errno
148 * is set accordingly on error.
149 */
150int udp_client::send(const char *msg, size_t size)
151{
152 return sendto(f_socket, msg, size, 0, f_addrinfo->ai_addr, f_addrinfo->ai_addrlen);
153}

159/** \brief Initialize a UDP server object.
160 *
161 * This function initializes a UDP server object making it ready to
162 * receive messages.
163 *
164 * The server address and port are specified in the constructor so
165 * if you need to receive messages from several different addresses
166 * and/or port, you'll have to create a server for each.
167 *
168 * The address is a string and it can represent an IPv4 or IPv6
169 * address.
170 *
171 * Note that this function calls connect() to connect the socket
172 * to the specified address. To accept data on different UDP addresses
173 * and ports, multiple UDP servers must be created.
174 *
175 * \note
176 * The socket is open in this process. If you fork() or exec() then the
177 * socket will be closed by the operating system.
178 *
179 * \warning
180 * We only make use of the first address found by getaddrinfo(). All
181 * the other addresses are ignored.
182 *
183 * \exception udp_client_server_runtime_error
184 * The udp_client_server_runtime_error exception is raised when the address
185 * and port combinaison cannot be resolved or if the socket cannot be
186 * opened.
187 *
188 * \param[in] addr The address we receive on.
189 * \param[in] port The port we receive from.
190 */
191udp_server::udp_server(const std::string& addr, int port)
192 : f_port(port)
193 , f_addr(addr)
194{
195 char decimal_port[16];
196 snprintf(decimal_port, sizeof(decimal_port), "%d", f_port);
197 decimal_port[sizeof(decimal_port) / sizeof(decimal_port[0]) - 1] = '\0';
198 struct addrinfo hints;
199 memset(&hints, 0, sizeof(hints));
200 hints.ai_family = AF_UNSPEC;
201 hints.ai_socktype = SOCK_DGRAM;
202 hints.ai_protocol = IPPROTO_UDP;
203 int r(getaddrinfo(addr.c_str(), decimal_port, &hints, &f_addrinfo));
204 if(r != 0 || f_addrinfo == NULL)
205 {
206 throw udp_client_server_runtime_error(("invalid address or port for UDP socket: \"" + addr + ":" + decimal_port + "\"").c_str());
207 }
208 f_socket = socket(f_addrinfo->ai_family, SOCK_DGRAM | SOCK_CLOEXEC, IPPROTO_UDP);
209 if(f_socket == -1)
210 {
211 freeaddrinfo(f_addrinfo);
212 throw udp_client_server_runtime_error(("could not create UDP socket for: \"" + addr + ":" + decimal_port + "\"").c_str());
213 }
214 r = bind(f_socket, f_addrinfo->ai_addr, f_addrinfo->ai_addrlen);
215 if(r != 0)
216 {
217 freeaddrinfo(f_addrinfo);
218 close(f_socket);
219 throw udp_client_server_runtime_error(("could not bind UDP socket with: \"" + addr + ":" + decimal_port + "\"").c_str());
220 }
221}

270/** \brief Wait on a message.
271 *
272 * This function waits until a message is received on this UDP server.
273 * There are no means to return from this function except by receiving
274 * a message. Remember that UDP does not have a connect state so whether
275 * another process quits does not change the status of this UDP server
276 * and thus it continues to wait forever.
277 *
278 * Note that you may change the type of socket by making it non-blocking
279 * (use the get_socket() to retrieve the socket identifier) in which
280 * case this function will not block if no message is available. Instead
281 * it returns immediately.
282 *
283 * \param[in] msg The buffer where the message is saved.
284 * \param[in] max_size The maximum size the message (i.e. size of the \p msg buffer.)
285 *
286 * \return The number of bytes read or -1 if an error occurs.
287 */
288int udp_server::recv(char *msg, size_t max_size)
289{
290 return ::recv(f_socket, msg, max_size, 0);
291}

293/** \brief Wait for data to come in.
294 *
295 * This function waits for a given amount of time for data to come in. If
296 * no data comes in after max_wait_ms, the function returns with -1 and
297 * errno set to EAGAIN.
298 *
299 * The socket is expected to be a blocking socket (the default,) although
300 * it is possible to setup the socket as non-blocking if necessary for
301 * some other reason.
302 *
303 * This function blocks for a maximum amount of time as defined by
304 * max_wait_ms. It may return sooner with an error or a message.
305 *
306 * \param[in] msg The buffer where the message will be saved.
307 * \param[in] max_size The size of the \p msg buffer in bytes.
308 * \param[in] max_wait_ms The maximum number of milliseconds to wait for a message.
309 *
310 * \return -1 if an error occurs or the function timed out, the number of bytes received otherwise.
311 */
312int udp_server::timed_recv(char *msg, size_t max_size, int max_wait_ms)
313{
314 fd_set s;
315 FD_ZERO(&s);
316 FD_SET(f_socket, &s);
317 struct timeval timeout;
318 timeout.tv_sec = max_wait_ms / 1000;
319 timeout.tv_usec = (max_wait_ms % 1000) * 1000;
320 int retval = select(f_socket + 1, &s, &s, &s, &timeout);
321 if(retval == -1)
322 {
323 // select() set errno accordingly
324 return -1;
325 }
326 if(retval > 0)
327 {
328 // our socket has data
329 return ::recv(f_socket, msg, max_size, 0);
330 }
331
332 // our socket has no data
333 errno = EAGAIN;
334 return -1;
335}
