28UDPChannel::UDPChannel(const std::string& address, int port)
29 : impl_(new Impl()) {
30 impl_->address = address;
31 impl_->port = port;
32
33 // Create UDP socket
34 impl_->socket_fd = socket(AF_INET, SOCK_DGRAM, 0);
35 if (impl_->socket_fd < 0) {
36 throw CommunicationException("Failed to create UDP socket");
37 }
38
39 // Set non-blocking
40 int flags = fcntl(impl_->socket_fd, F_GETFL, 0);
41 fcntl(impl_->socket_fd, F_SETFL, flags | O_NONBLOCK);
42
43 // Bind socket
44 struct sockaddr_in addr;
45 addr.sin_family = AF_INET;
46 addr.sin_port = htons(port);
47 addr.sin_addr.s_addr = inet_addr(address.c_str());
48
49 if (bind(impl_->socket_fd, (struct sockaddr*)&addr, sizeof(addr)) < 0) {
50 ::close(impl_->socket_fd);
51 throw CommunicationException("Failed to bind UDP socket");
52 }
53
54 impl_->active = true;
55 LOG_INFO("UDP channel initialized on " + address + ":" + std::to_string(port));
56}

62bool UDPChannel::send(const ACLMessage& message, const std::string& destination) {
63 if (!impl_->active) return false;
64
65 try {
66 // TODO: Serialize ACLMessage to string
67 std::string msg_str = "ACL message"; // Placeholder
68
69 // Parse destination (format: "host:port")
70 size_t colon_pos = destination.find(':');
71 if (colon_pos == std::string::npos) {
72 LOG_ERROR("Invalid destination format: " + destination);
73 return false;
74 }
75
76 std::string host = destination.substr(0, colon_pos);
77 int port = std::stoi(destination.substr(colon_pos + 1));
78
79 struct sockaddr_in dest_addr;
80 dest_addr.sin_family = AF_INET;
81 dest_addr.sin_port = htons(port);
82 dest_addr.sin_addr.s_addr = inet_addr(host.c_str());
83
84 ssize_t sent = sendto(impl_->socket_fd, msg_str.c_str(), msg_str.length(),
85 0, (struct sockaddr*)&dest_addr, sizeof(dest_addr));
86
87 return sent >= 0;
88
89 } catch (const std::exception& e) {
90 LOG_ERROR(std::string("UDP send failed: ") + e.what());
91 return false;
92 }
