305 //std::cout << "send signal to : " << this_agent->chldpid << std::endl;
306 }
307
308 switch (act) {
309 case Agent::AGENT_ACTIVE:
310 if (isLocal) {
311 std::unique_lock < std::mutex > lck(mtx);
312 action_to_do = Agent::AGENT_ACTIVE;
313 cv.notify_all();
314 } else {
315 sigqueue(chldpid, SIG_AGENT_ACTIVE, sval);

318 case Agent::AGENT_SUSPENDED:
319 if (isLocal) {
320 std::unique_lock < std::mutex > lck(mtx);
321 action_to_do = Agent::AGENT_SUSPENDED;
322 cv.notify_all();
323 } else {
324 sigqueue(chldpid, SIG_AGENT_SUSPEND, sval);
325 }
326 break;
327 case Agent::AGENT_TRANSIT:
328 if (isLocal) {
329 std::unique_lock < std::mutex > lck(mtx);
330 action_to_do = Agent::AGENT_TRANSIT;
331 cv.notify_all();
332 } else {
333 sigqueue(chldpid, SIG_AGENT_TRANSIT, sval);
334 }
335 break;
336 case Agent::AGENT_WAITING:
337 if (isLocal) {
338 std::unique_lock < std::mutex > lck(mtx);
339 action_to_do = Agent::AGENT_WAITING;
340 cv.notify_all();
341 } else {
342
343 sigqueue(chldpid, SIG_AGENT_WAIT, sval);
344 }
345 break;
346 case Agent::AGENT_DELETED:
347 if (isLocal) {
348 // Internal deletion - graceful shutdown
349 pid = boost::lexical_cast<std::string>(chldpid);
350 this->sendMsgMonitor("Stop agent PID : " + pid);
351
352 // First try graceful shutdown with SIGTERM
353 if (kill(chldpid, SIGTERM) == 0) {
354 // Wait briefly for graceful shutdown
355 int status;
356 pid_t result = waitpid(chldpid, &status, WNOHANG);
357
358 if (result == 0) {
359 // Process still running, wait a bit more
360 usleep(100000); // 100ms
361 result = waitpid(chldpid, &status, WNOHANG);
362 }
363
364 // If still running, force kill
365 if (result == 0) {
366 std::cerr << "Agent " << chldpid << " did not respond to SIGTERM, forcing SIGKILL" << std::endl;
367 kill(chldpid, SIGKILL);
368 waitpid(chldpid, &status, 0); // Wait for cleanup
369 }
370 } else {
371 perror("Failed to send termination signal");
372 }
373
374 // Clean up message queue
375 char* mq_name = this->get_msg_queue_name();
376 mqd_t mq = mq_open(mq_name, O_RDONLY | O_NONBLOCK);
377 if (mq != (mqd_t)-1) {
378 mq_close(mq);
379 mq_unlink(mq_name);
380 }
381 delete[] mq_name;
382 } else {
383 // External deletion request
384 sigqueue(chldpid, SIG_AGENT_DELETE, sval);
385 }
386 break;
387 case Agent::AGENT_WAKING:
388 if (isLocal) {
389 std::unique_lock < std::mutex > lck(mtx);
390 action_to_do = Agent::AGENT_WAKING;
391 cv.notify_all();
392 } else {
393 sigqueue(chldpid, SIG_AGENT_WAKE, sval);
394 }
395
396 break;
397 default:
398 break;
399 }
