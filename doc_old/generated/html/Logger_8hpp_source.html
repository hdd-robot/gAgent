75 /**
76 * @brief Log a message
77 * @param level Severity level
78 * @param message Message to log
79 * @param source Source component (optional)
80 */
81 void log(LogLevel level, const std::string& message,
82 const std::string& source = "") {
83 if (level < min_level_) {
84 return;
85 }
86
87 std::lock_guard<std::mutex> lock(mutex_);
88
89 std::ostringstream oss;
90 oss << getTimestamp() << " "
91 << "[" << levelToString(level) << "] ";
92
93 if (!source.empty()) {
94 oss << "[" << source << "] ";
95 }
96
97 oss << message << std::endl;
98
99 std::string formatted = oss.str();
100 std::cout << formatted;
101
102 if (log_file_.is_open()) {
103 log_file_ << formatted;
104 log_file_.flush();
105 }
106 }

109 * @brief Close log file

113 if (log_file_.is_open()) {

150 std::string levelToString(LogLevel level) const {
151 switch (level) {
152 case LogLevel::DEBUG: return "DEBUG";
153 case LogLevel::INFO: return "INFO ";

161 std::string getTimestamp() const {
162 auto now = std::chrono::system_clock::now();
163 auto time = std::chrono::system_clock::to_time_t(now);
164 auto ms = std::chrono::duration_cast<std::chrono::milliseconds>(
165 now.time_since_epoch()) % 1000;
166
167 std::ostringstream oss;
168 oss << std::put_time(std::localtime(&time), "%Y-%m-%d %H:%M:%S")
169 << "." << std::setfill('0') << std::setw(3) << ms.count();
170 return oss.str();
171 }
