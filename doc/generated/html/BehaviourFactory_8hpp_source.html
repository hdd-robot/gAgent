20/**
21 * @brief Factory for creating Behaviour instances
22 *
23 * Implements the Factory pattern to create behaviour objects dynamically.
24 * Supports registration of custom behaviour types.
25 *
26 * Example usage:
27 * @code
28 * auto factory = BehaviourFactory::getInstance();
29 * factory.registerBehaviour("MyBehaviour",
30 * [](Agent* ag) { return std::make_unique<MyBehaviour>(ag); });
31 *
32 * auto behaviour = factory.create("MyBehaviour", agent);
33 * @endcode
34 */
35class BehaviourFactory {
36public:
37 using BehaviourCreator = std::function<std::unique_ptr<Behaviour>(Agent*)>;
38
39 /**
40 * @brief Get singleton instance
41 * @return Reference to factory instance
42 */
43 static BehaviourFactory& getInstance() {
44 static BehaviourFactory instance;
45 return instance;
46 }
47
48 /**
49 * @brief Register a new behaviour type
50 * @param name Type name identifier
51 * @param creator Function to create the behaviour
52 */
53 void registerBehaviour(const std::string& name, BehaviourCreator creator) {
54 creators_[name] = creator;
55 }
56
57 /**
58 * @brief Create a behaviour instance
59 * @param name Type name
60 * @param agent Agent owner
61 * @return Unique pointer to created behaviour
62 * @throw BehaviourException if type not found
63 */
64 std::unique_ptr<Behaviour> create(const std::string& name, Agent* agent) {
65 auto it = creators_.find(name);
66 if (it == creators_.end()) {
67 throw BehaviourException("Unknown behaviour type: " + name);
68 }
69 return it->second(agent);
70 }
71
72 /**
73 * @brief Check if behaviour type is registered
74 * @param name Type name
75 * @return true if registered
76 */
77 bool isRegistered(const std::string& name) const {
78 return creators_.find(name) != creators_.end();
79 }
80
81 // Delete copy/move
82 BehaviourFactory(const BehaviourFactory&) = delete;
83 BehaviourFactory& operator=(const BehaviourFactory&) = delete;
84
85private:
86 BehaviourFactory() = default;
87 std::map<std::string, BehaviourCreator> creators_;
88};
