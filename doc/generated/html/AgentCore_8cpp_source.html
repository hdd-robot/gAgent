70void AgentCore::cache_signal_handler(int signal_num) {
71 const char* signal_name = "UNKNOWN";
72 switch(signal_num) {
73 case SIGINT: signal_name = "SIGINT"; break;
74 case SIGTERM: signal_name = "SIGTERM"; break;
75 case SIGQUIT: signal_name = "SIGQUIT"; break;
76 }
77
78 std::cout << "Caught signal " << signal_name << " (" << signal_num
79 << "), shutting down all agents..." << std::endl;
80
81 // Get current process group ID
82 pid_t pgid = getpgrp();
83
84 // Send SIGTERM to all processes in the process group
85 // Using negative PID sends to the entire process group
86 if (kill(-pgid, SIGTERM) == -1) {
87 perror("Failed to send SIGTERM to process group");
88 }
89
90 // Give processes time to clean up
91 sleep(1);
92
93 // Force kill any remaining processes
94 kill(-pgid, SIGKILL);
95
96 exit(EXIT_SUCCESS);
97}
