51 attr.mq_curmsgs = 0;

54 perror("mq_open ");

63 int ret;
64 while (true) {
65 //std::cout << "begin receve from agents " << std::endl;
66 ret = mq_receive(mq, buffer, taille, NULL);
67 if(ret < 0){
68 perror("mq_receive ");
69 }
70

72 m.clear();
73
74 if (MessageParser::parsekeyValSeq(sbuffer,m)){

164void gagent::Environnement::make_agent() {
165
166 std::map<std::string, std::map<std::string, std::string>>::iterator it;
167 std::map<std::string, std::string>::iterator it2;
168
169 for (int i = 0; i < this->list_visual_agents.size(); i++) {
170 delete this->list_visual_agents[i];
171 }
172
173 this->list_visual_agents.clear();
174
175 for (it = this->list_attr.begin(); it != this->list_attr.end(); it++) {
176 gagent::VisualAgent* v_agent = new gagent::VisualAgent();
177 //std::cout << "=> enregistrement : " << it->first << std::endl;
178 for (it2 = it->second.begin(); it2 != it->second.end(); it2++) {
179 //std::cout << "velue : " << it2->first << " : " << it2->second << std::endl;
180 if(it2->first == this->id) { v_agent->id = it2->second ;}
181 else if(it2->first == this->name) { v_agent->name = it2->second ;}
182 else if(it2->first == this->pos_x) {try{ v_agent->pos_x= std::stof(it2->second);}catch(...){v_agent->pos_x=0;}}
183 else if(it2->first == this->pos_y) {try{ v_agent->pos_y= std::stof(it2->second);}catch(...){v_agent->pos_y=0;}}
184 else if(it2->first == this->val) { v_agent->val = it2->second ;}
185 else if(it2->first == this->shape) { v_agent->shape = it2->second ;}
186 else if(it2->first == this->color) { v_agent->color = it2->second ;}
187 else if(it2->first == this->pattern){ v_agent->pattern = it2->second ;}
188 else if(it2->first == this->size) {try{v_agent->size = std::stof(it2->second);}catch(...){v_agent->size=0;}}
189 else if(it2->first == this->size_x) {try{v_agent->size_x= std::stof(it2->second);}catch(...){v_agent->size_x=0;}}
190 else if(it2->first == this->size_y) {try{v_agent->size_y= std::stof(it2->second);}catch(...){v_agent->size_y=0;}}
191 else if(it2->first == this->size_z) {try{v_agent->size_z= std::stof(it2->second);}catch(...){v_agent->size_z=0;}}
192 else { v_agent->vals.push_back(it2->second);}
193 }
194
195 if (v_agent->id ==""){
196 delete (v_agent);
197 continue;
198 }
199
200 if (v_agent->shape =="") { v_agent->shape = "circle";}
201 if (v_agent->size == 0) { v_agent->size = 5.0;}
202
203 this->list_visual_agents.push_back(v_agent);
204
205 }
